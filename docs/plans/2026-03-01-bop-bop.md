# bop-bop: Live Agent Sessions via Jobcard + Zellij Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Make `.jobcard` bundles the live terminal — click a card in Finder Quick Look → `bop://` URL → resumable zellij session containing the running agent.

**Architecture:** Add `zellij_session` to `meta.json`; `bop_bop.nu` creates the card + JJ workspace + named zellij session in one command; the `sh.bop` macOS host app registers the `bop://` URL scheme and routes `/session` to `zellij attach`; Quick Look renders an "Attach" button when the session field is present.

**Tech Stack:** Rust (bop-core Meta struct), Nushell, Zellij, Swift/SwiftUI (Quick Look + URL handler), jj.

## Task 3: `scripts/bop_bop.nu` — bootstrap script

**Files:**
- Create: `scripts/bop_bop.nu`

**Step 1: Write the script**

```sh
#!/usr/bin/env nu
set -euo pipefail
setopt NULL_GLOB

ROOT=${0:A:h:h}
BOP="${ROOT}/target/debug/bop"

if [[ $# -lt 1 ]]; then
  print "Usage: bop_bop.nu <goal description>" >&2
  exit 1
fi

goal="$*"

# Slugify goal → card id
id="${goal:l}"           # lowercase
id="${id// /-}"          # spaces → hyphens
id="${id//[^a-z0-9-]/}" # strip non-url chars
id="${id:0:40}"          # max 40 chars
id="bop-${id}"

session="bop-${id}"

print "▶ Creating card: ${id}"
"${BOP}" new implement "${id}" 2>/dev/null || true

# Locate the card
card_dir=""
for state in pending running done; do
  candidate="${ROOT}/.cards/${state}/${id}.jobcard"
  [[ -d "$candidate" ]] && card_dir="$candidate" && break
done

if [[ -z "$card_dir" ]]; then
  print "ERROR: card not found after creation" >&2
  exit 1
fi

# Write goal into spec.md
print "# ${goal}\n\nCreated by bop_bop.nu." >> "${card_dir}/spec.md"

# Write zellij_session into meta.json (use python3 for safe JSON merge)
python3 - "${card_dir}/meta.json" "${session}" <<'PY'
import json, sys
path, session = sys.argv[1], sys.argv[2]
meta = json.loads(open(path).read())
meta["zellij_session"] = session
meta["zellij_pane"] = "1"
open(path, "w").write(json.dumps(meta, indent=2, ensure_ascii=False))
print(f"  zellij_session: {session}")
PY

# Budget-aware agent routing
# Cost order: ollama (free) → opencode ($) → codex ($) → claude ($$)
PROVIDER="mock"
ADAPTER="${ROOT}/adapters/mock.nu"

if command -v ollama &>/dev/null; then
  PROVIDER="ollama"; ADAPTER="${ROOT}/adapters/ollama-local.nu"
fi
if command -v opencode &>/dev/null; then
  PROVIDER="opencode"; ADAPTER="${ROOT}/adapters/opencode.nu"
fi
if command -v codex &>/dev/null; then
  PROVIDER="codex"; ADAPTER="${ROOT}/adapters/codex.nu"
fi
if command -v claude &>/dev/null; then
  PROVIDER="claude"; ADAPTER="${ROOT}/adapters/claude.nu"
fi

print "  provider: ${PROVIDER}"
print "  session:  ${session}"
print ""
print "▶ bop://card/${id}/session"
print ""

# Create JJ workspace if jj is available
if command -v jj &>/dev/null && jj root &>/dev/null 2>&1; then
  worktree_dir="${ROOT}/.worktrees/${id}"
  jj workspace add "${worktree_dir}" 2>/dev/null || true
  python3 - "${card_dir}/meta.json" "${id}" "${worktree_dir}" <<'PY'
import json, sys
path, branch, ws = sys.argv[1], sys.argv[2], sys.argv[3]
meta = json.loads(open(path).read())
meta["worktree_branch"] = f"job/{branch}"
meta["workspace_path"] = ws
open(path, "w").write(json.dumps(meta, indent=2, ensure_ascii=False))
PY
fi

# Launch or attach resumable zellij session with dispatcher inside
if command -v zellij &>/dev/null; then
  zellij attach "${session}" 2>/dev/null || \
    zellij -s "${session}" -- \
      "${BOP}" dispatcher \
        --adapter "${ADAPTER}" \
        --once
else
  # No zellij: run dispatcher inline
  "${BOP}" dispatcher --adapter "${ADAPTER}" --once
fi
```

## Task 5: `sh.bop` host app — `bop://` URL scheme

**Files:**
- Modify: `macos/JobCardHost/Info.plist` — register URL scheme
- Modify: `macos/JobCardHost/AppDelegate.swift` (or equivalent) — handle URL

**Step 1: Register `bop` URL scheme in Info.plist**

Add to `macos/JobCardHost/Info.plist`:

```xml
<key>CFBundleURLTypes</key>
<array>
    <dict>
        <key>CFBundleURLName</key>
        <string>sh.bop.url</string>
        <key>CFBundleURLSchemes</key>
        <array>
            <string>bop</string>
        </array>
    </dict>
</array>
```

**Step 2: Handle URL in AppDelegate**

In `AppDelegate.swift` (or `@main` App struct), add:

```swift
func application(_ application: NSApplication,
                 open urls: [URL]) {
    for url in urls {
        handleBopURL(url)
    }
}

private func handleBopURL(_ url: URL) {
    // bop://card/<id>/session  → zellij attach bop-<id>
    guard url.scheme == "bop",
          url.host == "card",
          let id = url.pathComponents.dropFirst().first else { return }

    let action = url.pathComponents.last ?? "session"

    switch action {
    case "session":
        let session = "bop-\(id)"
        // Resumable: attach existing or create new
        let script = """
        zellij attach '\(session)' 2>/dev/null || zellij -s '\(session)'
        """
        runInTerminal(script)
    case "spec":
        openInEditor(cardPath(id: id, file: "spec.md"))
    default:
        NSWorkspace.shared.open(cardURL(id: id))
    }
}

private func runInTerminal(_ script: String) {
    // Open in user's default terminal
    let args = ["-e", "nu", "-c", script]
    NSWorkspace.shared.open(
        URL(fileURLWithPath: "/Applications/Ghostty.app"),
        configuration: NSWorkspace.OpenConfiguration()
    )
    // Fallback: use open -a Terminal
    Process.launchedProcess(
        launchPath: "/usr/bin/open",
        arguments: ["-a", "Terminal", "--args", "nu", "-c", script]
    )
}
```

Note: Detect terminal from `$TERM_PROGRAM` or user preference. Ghostty first, fallback to Terminal.app.

The `zellij_session` field in meta.json is the source of truth.
Quick Look renders a live "Attach" button when the field is present.
